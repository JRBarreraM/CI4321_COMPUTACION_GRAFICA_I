set(PTHREAD_ROOT_DIR ${CMAKE_CURRENT_SOURCE_DIR} CACHE PATH "" FORCE)

set(PTHREAD_HEADER_DIR ${PTHREAD_ROOT_DIR}/include)
file(GLOB PTHREAD_DLLS ${PTHREAD_ROOT_DIR}/bin/*.dll)

include_directories(${PTHREAD_HEADER_DIR})

add_library(pthread SHARED IMPORTED)

set(PTHREAD_LIBRARY_DIRS "${PTHREAD_ROOT_DIR}/lib" CACHE PATH "" FORCE)
set(PTHREAD_LIBRARIES pthreadVC2 CACHE PATH "" FORCE)
set(PTHREAD_LIBRARIES pthreadVC2d CACHE PATH "" FORCE)
set(PTHREAD_INCLUDE_DIRS ${PTHREAD_HEADER_DIR} CACHE PATH "" FORCE)
set(PTHREAD_RUNTIMELIBS ${PTHREAD_DLLS} CACHE PATH "" FORCE)
